function potenciaSemCorrente(tensao, resistencia){
    return (tensao * tensao) / resistencia
}

function potenciaSemResistencia(tensao, corrente){
    return tensao * corrente
}

function potenciaSemTensao(resistencia, corrente){
    return resistencia * (corrente * corrente)
}

function principal(){
    let resistencia = 25
    let tensao = 220
    let corrente = 8.8

    console.log(`Descobrindo a potência 'W' a partir da tensão 'V' e resistência 'R': ${potenciaSemCorrente(tensao, resistencia)}`)
    console.log(`Descobrindo a potência 'W' a partir da tensão 'V' e corrente 'I': ${potenciaSemResistencia(tensao, corrente)}`)
    console.log(`Descobrindo a potência 'W' a partir da resistência 'R' e corrente 'I': ${potenciaSemTensao(resistencia, corrente)}`)
}
